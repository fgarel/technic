Manipulation de la base osm
---------------------------

-- Selection des rues et des places
select *
from ways
where ( "tags" -> 'name' is not null
    and "tags" -> 'amenity' = 'parking')
  or (  "tags" -> 'name' is not null 
    and "tags" -> 'highway' is not null)
limit 100;
